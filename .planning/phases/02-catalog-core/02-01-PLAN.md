```typescript
import slugify from 'slugify'
import fs from 'fs/promises'
import path from 'path'
import { getLibraryDir } from './library-service'

export function generateSlug(title: string): string {
  return slugify(title, {
    lower: true,
    strict: true,
    locale: 'pt',
  })
}

export async function resolveSlugCollision(baseSlug: string): Promise<string> {
  const dir = getLibraryDir()
  let slug = baseSlug
  let counter = 1
  while (true) {
    try {
      await fs.access(path.join(dir, `${slug}.md`))
      counter++
      slug = `${baseSlug}-${counter}`
    } catch {
      return slug
    }
  }
}
```

Key details:
- `slugify` with `strict: true` strips all characters except alphanumeric and the replacement character, which prevents path traversal by design (no `..`, `/`, `\` can survive)
- `locale: 'pt'` handles Portuguese accent transliteration (a with accent -> a)
- Counter starts at 1, increments before assignment, so collisions produce -2, -3, etc. (per D-10)
- Import `getLibraryDir` from `./library-service` (same module, consistent with project pattern)

**writeBook function:**
```typescript
export async function writeBook(input: WriteBookInput): Promise<{ slug: string }> {
  const dir = getLibraryDir()
  const baseSlug = generateSlug(input.title)
  const slug = await resolveSlugCollision(baseSlug)
  const filepath = path.join(dir, `${slug}.md`)

  const { notes, ...rest } = input
  const frontmatter: Record<string, unknown> = {
    ...rest,
    added_at: new Date().toISOString().split('T')[0],
  }
  const content = matter.stringify(notes ?? '', frontmatter)
  await fs.writeFile(filepath, content, 'utf-8')

  return { slug }
}
```

**updateBook function:**
```typescript
export async function updateBook(
  slug: string,
  updates: Partial<WriteBookInput>
): Promise<void> {
  const dir = getLibraryDir()
  const filepath = path.join(dir, `${slug}.md`)

  const raw = await fs.readFile(filepath, 'utf-8')
  const { data, content } = matter(raw, SAFE_MATTER_OPTIONS)

  const { notes, ...frontmatterUpdates } = updates
  const newData = { ...data, ...frontmatterUpdates }
  const newContent = notes !== undefined ? notes : content

  const output = matter.stringify(newContent, newData)
  await fs.writeFile(filepath, output, 'utf-8')
}
```

**deleteBook function:**
```typescript
export async function deleteBook(slug: string): Promise<void> {
  const dir = getLibraryDir()
  const filepath = path.join(dir, `${slug}.md`)
  await fs.unlink(filepath)
}
```

Key details:
- `writeBook` uses `matter.stringify()` from gray-matter (already imported) to produce valid YAML frontmatter
- `updateBook` reads with `SAFE_MATTER_OPTIONS` (CVE mitigation) before rewriting -- per PATTERNS.md
- `deleteBook` lets `fs.unlink` throw on missing file (explicit error, per PATTERNS.md convention)
- `added_at` is set to ISO date string `YYYY-MM-DD` on creation only
- Export `WriteBookInput` interface so API routes can import it

| Boundary | Description |
|----------|-------------|
| slug input -> filesystem | Book title from user input becomes a filename on disk |
| external updates -> matter parse | Manually edited .md files are re-parsed by updateBook |

| Threat ID | Category | Component | Disposition | Mitigation Plan |
