**Create `src/lib/api/google-books.ts`:**

```typescript
const GOOGLE_BOOKS_API = 'https://www.googleapis.com/books/v1/volumes'

export interface BookSearchResult {
  title: string
  authors: string[]
  isbn?: string
  synopsis?: string
  cover?: string
  genre?: string
  year?: number
}

export async function searchGoogleBooks(
  query: string,
  maxResults = 5
): Promise<BookSearchResult[]> {
  const params = new URLSearchParams({
    q: query,
    maxResults: String(maxResults),
    langRestrict: 'pt',
  })

  const apiKey = process.env.GOOGLE_BOOKS_API_KEY
  if (apiKey) {
    params.set('key', apiKey)
  }

  const res = await fetch(`${GOOGLE_BOOKS_API}?${params}`)
  if (!res.ok) {
    throw new Error(`[GoogleBooks] API error: ${res.status}`)
  }

  const data = await res.json()
  return (data.items ?? []).map((item: any) => {
    const v = item.volumeInfo ?? {}
    const isbn = v.industryIdentifiers?.find(
      (id: any) => id.type === 'ISBN_13' || id.type === 'ISBN_10'
    )
    return {
      title: v.title ?? '',
      authors: v.authors ?? [],
      isbn: isbn?.identifier,
      synopsis: v.description,
      cover: v.imageLinks?.thumbnail?.replace('http://', 'https://'),
      genre: v.categories?.[0],
      year: v.publishedDate
        ? parseInt(v.publishedDate.slice(0, 4), 10) || undefined
        : undefined,
    } satisfies BookSearchResult
  })
}
```

**Create `src/lib/api/open-library.ts`:**

```typescript
import type { BookSearchResult } from './google-books'

const OPEN_LIBRARY_API = 'https://openlibrary.org/search.json'

export async function searchOpenLibrary(
  query: string,
  limit = 5
): Promise<BookSearchResult[]> {
  const params = new URLSearchParams({
    q: query,
    limit: String(limit),
    fields: 'title,author_name,first_publish_year,cover_i,isbn',
  })

  const res = await fetch(`${OPEN_LIBRARY_API}?${params}`, {
    headers: { 'User-Agent': 'DonaFlora/1.0 (personal book catalog)' },
  })
  if (!res.ok) {
    throw new Error(`[OpenLibrary] API error: ${res.status}`)
  }

  const data = await res.json()
  return (data.docs ?? []).map((doc: any) => ({
    title: doc.title ?? '',
    authors: doc.author_name ?? [],
    isbn: doc.isbn?.[0],
    cover: doc.cover_i
      ? `https://covers.openlibrary.org/b/id/${doc.cover_i}-L.jpg`
      : undefined,
    year: doc.first_publish_year,
  } satisfies BookSearchResult))
}
```

**Create `src/lib/markdown.ts`:**
```typescript
import { unified } from 'unified'
import remarkParse from 'remark-parse'
import remarkRehype from 'remark-rehype'
import rehypeSanitize from 'rehype-sanitize'
import rehypeStringify from 'rehype-stringify'

export async function renderMarkdown(markdown: string): Promise<string> {
  if (!markdown.trim()) return ''
  const file = await unified()
    .use(remarkParse)
    .use(remarkRehype)
    .use(rehypeSanitize)
    .use(rehypeStringify)
    .process(markdown)
  return String(file)
}
```

IMPORTANT: Do NOT use `remark-html` -- it is in maintenance mode (per RESEARCH.md). Use the `remark-rehype` + `rehype-stringify` pipeline. `rehype-sanitize` is mandatory for XSS prevention in user-authored notes.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client -> API route -> external API | User search query forwarded to Google Books/Open Library |
| external API response -> BookSearchResult | Untrusted JSON mapped to typed interface |
| user notes (Markdown) -> rendered HTML | User-authored content rendered in browser |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-04 | Information Disclosure | google-books.ts | mitigate | API key read from `process.env.GOOGLE_BOOKS_API_KEY` (no `NEXT_PUBLIC_` prefix). Module runs only on server via API routes. |
| T-02-05 | Tampering | markdown.ts (XSS) | mitigate | `rehype-sanitize` strips script tags, event handlers, and dangerous HTML. Test confirms `<script>` and `onerror` are removed. |
| T-02-06 | Denial of Service | open-library.ts | accept | Open Library rate limits (1-3 req/s). Debounce in UI layer (Plan 05) reduces request volume. Single-user app, low risk. |
| T-02-07 | Spoofing | google-books.ts / open-library.ts | accept | External API responses are untrusted but only used for display metadata. No security-critical decisions based on API data. |
</threat_model>
